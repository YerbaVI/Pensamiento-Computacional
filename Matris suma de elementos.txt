// Matris suma de elementos
#include <iostream>
using namespace std;
int main (){
    int i,j,n,a;
    cin >> n;
    int A[n][n];
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            cin >> A[i][j];
        }
    }
    a=0;
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            a=a+A[i][j];
        }
    }
    cout << a;
    return 0;
}
