//evaluando una formula gigante
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;
int main() {
    double x, y, z;
    cin >> x >> y >> z;
    double a = ((2 * x + y) / z) * (pow(y, 3) - z);
    double b = ((x + 2 * y + 3 * z) / (z - 2 * y - 3 * x)) + (pow(x, 2) + pow(z, 2));
    double R = a / b;
    cout << fixed << setprecision(6) << R << endl;
    return 0;
}
